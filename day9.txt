6. Write a program in Java to display n terms of natural numbers and their sum
Ans6:--package Basic_Codes;

import java.util.Scanner;

public class natural_num {

	public static void main(String[] args) 
	{
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter the n natural numbers");
		
		int n=sc.nextInt();
		int sum=0;
		for(int i=1;i<=n;i++)
		{
		 sum = sum+i;	
		}
		System.out.println("Sum is "+sum);
	}

}
