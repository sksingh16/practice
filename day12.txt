	      {

	         }
	      }
	      catch(IOException e)
	      {
	         System.out.println("Something went wrong!");
	         System.out.print("Exception: " +e);

8. Write a program to count characters, words and lines in a text file.

Ans8:--package day12;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class word_char_count {

	public static void main(String[] args) {
		
		        BufferedReader br = null;
		        int char_count = 0;
		        int word_Count = 0;
		        int line_Count = 0;
		         
		        try
		        {
		            br = new BufferedReader(new FileReader("E:\\Sample2\\demo_2.txt"));
		            String s = br.readLine();
		            while (s != null)
		            {
		                line_Count++;
		                
		                String[] words = s.split(" ");
		                word_Count = word_Count + words.length;
		                 
		                for (String word : words)
		                {
		                    char_count = char_count + word.length();
		                }
		                
		                s = br.readLine();
		            }
		             System.out.println("Number of character in file : "+char_count);
		             System.out.println("Number of words in a file : "+word_Count);
		             System.out.println("Number of lines in file : "+line_Count);
		        }
		        
		        catch (IOException e)
		        {
		           System.out.println("Error occured");
		        }
		        
		        finally
		        {
		            try
		            {
		                br.close();          
		            }
		            catch (IOException e)
		            {
		            	 System.out.println("Error occured");
		            }
		        }
		    
	}

}

18. Write a program to convert uppercase to lowercase character and vice versa in a text file.

Ans18:--package day12;

import java.io.*;

public class converting_in_uppernlowercase {
    public static void main (String[] args) throws IOException 
    {
          
    					String s = "E:\\Sample2\\abc.txt";
    	               
    	    try {
    	    	
    	    	FileReader fr = new FileReader(s);
    	    	int i;
    	    	String data="";
    	    	
    	    	while((i=fr.read())!= -1) 
    	    	{
    	    		char ch = (char)i;
    	    		if(ch>='a' && ch<='z') 
    	    		{
    	    			ch = Character.toUpperCase(ch);
    	    			
    	    		}
    	    		else if(ch>='A' && ch<='Z') 
    	    		{
    	    			ch = Character.toLowerCase(ch);
    	    		}
    	    		data = data+ch;
    	    		
    	    	}
    	    	System.out.println(data);
    	    	fr.close();
    	    	
    	    	FileWriter fw = new FileWriter(s);
    	    	fw.write(data);
    	    	fw.flush();
    	    	fw.close();
    	    	
    	    	
    	    }
    	    catch(Exception e) 
    	    {
    	    	System.out.println("Error");
    	    	
    	    }
    	
    }   
     

}
